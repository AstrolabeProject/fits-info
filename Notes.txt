# build fits-info image
dk build -t fits-info .

## Run examples:

# show help info:
dk run --rm fits-info

# explicitly call for help info:
dk run --rm fits-info --help

# verify all FITS files in /data and below:
dk run --rm --name fits-info -v /data/UAZ/iSchool/work/astrolabe_vm/sample-data/:/data fits-info --verify /data

# show info on a single FITS file:
dk run --rm --name fits-info -v /data/UAZ/iSchool/work/astrolabe_vm/sample-data/:/data fits-info --info /data/Hunter/cvnidwabcut.fits

# extract metadata for all FITS files in given directory:
dk run --rm --name fits-info -v /data/UAZ/iSchool/work/astrolabe_vm/sample-data/:/data fits-info --metadata /data/Hunter

# extract metadata for all FITS files in given directory, using an alternate keyset:
dk run --rm --name fits-info -v /data/UAZ/iSchool/work/astrolabe_vm/sample-data/:/data fits-info --keyfile /tmp/md-keys.txt --metadata /data/Hunter

# no FITS files in this directory to extract metadata from:
dk run --rm --name fits-info -v /data/UAZ/iSchool/work/astrolabe_vm/sample-data/:/data fits-info --metadata /data/MSMA
